import random
import pandas as pd 

def csv_create():
    responses = {
    "hello": ["Hello there!", "Hi, how are you?"],
    "how are you": ["I'm doing well, thanks for asking!", "I'm a bot, so I don't have feelings, but I'm functioning normally."],
    "what is your name": ["My name is ChatBot!", "I'm ChatBot, nice to meet you."],
    "bye": ["Goodbye!", "See you later!"],
    "default": ["I'm sorry, I didn't understand what you said. Can you please rephrase that?", "Could you try asking me something else?"]
    }
    dataframe = pd.Dataframe(responses)
    dataframe.to_csv('dataset.csv', index = False)
def csv_read():
    dataframe = pd.read_csv('dataset.csv')
    responses = dataframe.to_dict('records')
    print(responses)
    return responses
responses = csv_read()
# Define a function to respond to user input
def respond(user_input):
    # Convert user input to lowercase
    user_input = user_input.lower()
    
    # Check if user input matches a key in the responses dictionary
    if user_input in responses:
        # If so, select a random response from the corresponding list
        bot_response = random.choice(responses[user_input])
    else:
        # If not, select a default response
        bot_response = random.choice(responses["default"])
    # Return the bot's response
    return bot_response

# Prompt the user for input and respond until the user enters "bye"
def conservation():
    while True:
        user_input = input("You: ")
        bot_response = respond(user_input)
        print("ChatBot:", bot_response)
        if user_input == "bye":
            break

int checksum(long number, int i)
{
    //Luhn's Algorithm (checksum)
    int numbercs1 = number % 10;
    //gets last digit (dividing by 10 to get remainder gives the digit itself)
    number /= 10;
    //moves to next digit (the second-last digit)
    int numbercs2 = subcheck(number);
    //numbercs1 and numbercs2 outside of loop to get the initial values so the following can be added on
    for (int r = 0; r < i; r++)
    {
        //loops through the entire card number
        number /= 10;
        //moves to next digit
        numbercs1 += number % 10;
        //gets digit for the add sum part of the checksum validation
        number /= 10;
        numbercs2 += subcheck(number);
        //gets digit for the 'multiply every other digit by 2' part of the checksum
    }
    int numbercs = numbercs1 + numbercs2;
    //adds the final numbers from add sum and multipled sum together
    if (numbercs % 10 == 0)
        //checks if the last digit is 0 (total modulo congruent 10 to 0)
    {
        return 1;
        //if it is, returns 1 (true)
    }
    else
    {
        return 0;
        //if not, returns 0 (false)
    }
}
int subcheck(long number)
{
    int numbercs2 = (number % 10) * 2;
    //multiples digit taken from number by 2

        //if the digit multipled is 10 or over, it becomes 2 digits
    {
        //so it needs to be split into individual digits
        int num1 = numbercs2 % 10;
        numbercs2 /= 10;
        //moves to next digit of the number multipled

        return num1 + num2;
        //returns sum of the split digits

    return numbercs2;
    //if it doesn't need to be split, returns multipled number as it is
}
